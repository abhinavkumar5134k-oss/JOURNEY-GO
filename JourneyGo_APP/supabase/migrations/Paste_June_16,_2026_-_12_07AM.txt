def get_business_info(category: str) -> str:
    category = category.lower().strip()
    json_path = os.path.join(os.path.dirname(__file__), "intents.json")
    try:
        with open(json_path, "r") as file:
            data = json.load(file)
        for intent in data.get("intents", []):
            if intent["tag"] == category:
                return intent["response"]
        return f"Error: Category '{category}' not found."
    except Exception as e:
        return str(e)

def get_weather_info(city: str) -> str:
    """Allows Trackmate AI to fetch real-time weather with improved accuracy."""
    # 1. Clean the input to remove accidental quotes or extra spaces
    clean_city = city.strip().replace("'", "").replace('"', "")
    
    # Debugging print statement for your terminal
    print(f"DEBUG: Weather tool attempting to fetch for: '{clean_city}'") 
    
    api_key = os.getenv("WEATHER_API_KEY", "fc64e8cbf84d5060f7d160fa3a63acaf")
    
    # 2. Append ',IN' to the city string to force the API to look in India
    # This prevents the API from returning weather in the wrong country/continent
    url = f"https://api.openweathermap.org/data/2.5/weather?q={clean_city},IN&appid={api_key}&units=metric"
    
    try:
        response = requests.get(url)
        if response.status_code == 200:
            data = response.json()
            temp = round(data["main"]["temp"])
            desc = data["weather"][0]["description"]
            return f"The current weather in {clean_city} is {temp}°C with {desc}."
        else:
            # If the specific name fails, it returns this error
            return f"Sorry, I couldn't find weather data for {clean_city}."
    except Exception as e:
        return "Weather service is currently unavailable."

CORE BEHAVIORS:

8. CRITICAL: You must always write a full sentence response for the user to read/hear. Do not output tags alone.
9. NEVER repeat a function call in the same response.
10. Output the tag and ONLY the result. DO NOT include the tag in your final text response to the user."""

@app.post("/chat")
async def handle_chat_session(payload: ChatPayload):
    try:
        user_text = payload.message
        if payload.audio_base64:
            audio_data = base64.b64decode(payload.audio_base64)
            audio_file = BytesIO(audio_data)
            audio_file.name = "input.webm"
            transcription = client.audio.transcriptions.create(file=audio_file, model="whisper-large-v3", prompt="Transcribe.")
            user_text = transcription.text

        history_manager = get_session_history(payload.session_id)
        messages = [{"role": "system", "content": SYSTEM_INSTRUCTIONS}]
        for msg in history_manager.messages:
            role = "assistant" if msg.__class__.__name__ == "AIMessage" else "user"
            messages.append({"role": role, "content": msg.content})
        messages.append({"role": "user", "content": user_text})
        
        response = client.chat.completions.create(model="llama-3.3-70b-versatile", messages=messages, temperature=0.0)
        raw_output = response.choices[0].message.content

        # --- MULTI-TOOL PARSING & CLEANING ---
        tool_matches = re.finditer(r'<function-(\w+)>(.*?)</function-\1>', raw_output, re.DOTALL)
        tool_results = []
        assistant_output = raw_output
        
        for tool_match in tool_matches:
            tool_name = tool_match.group(1)
            tool_args = json.loads(tool_match.group(2).strip())
            
            if tool_name == "locate_place":
                tool_results.append(tool_args.get("place_name"))
            elif tool_name in AVAILABLE_TOOLS:
                result = AVAILABLE_TOOLS[tool_name](**tool_args)
                assistant_output += f"\n\nResult: {result}"

        # Clean the final output by removing ALL tags for the user
        clean_output = re.sub(r'<function-\w+>.*?</function-\w+>', '', assistant_output, flags=re.DOTALL).strip()

        # FALLBACK: If the AI outputted tags but forgot to write text, provide a default
        if not clean_output:
            clean_output = "I have processed your request."

        history_manager.add_user_message(user_text)
        history_manager.add_ai_message(clean_output)

        # --- AUDIO GENERATION (Always run this) ---
        audio_b64 = None
        if clean_output and clean_output.strip():
            try:
                tts = gTTS(text=clean_output, lang='en')
                fp = BytesIO()
                tts.write_to_fp(fp)
                fp.seek(0)
                audio_b64 = base64.b64encode(fp.read()).decode('utf-8')
            except Exception as e:
                print(f"TTS Error: {e}")
        
        # --- RETURN STATEMENT RESTORED HERE ---
        return {"response": clean_output, "tool_triggered": tool_results, "audio": audio_b64}

    except Exception as e:
        print(f"Error: {e}")
        raise HTTPException(status_code=500, detail=str(e))

function appendMessage(text, sender) {

    const messageDiv = document.createElement("div");
    messageDiv.classList.add("message", sender);
    messageDiv.textContent = text;
    chatMessages.appendChild(messageDiv);

function playAudio(base64Data) {
    if (!base64Data) return;
    try {
        const audioUrl = `data:audio/mp3;base64,${base64Data}`;
        const audio = new Audio(audioUrl);
        audio.play().catch(err => console.error("Audio playback interrupted:", err));
    } catch (error) {
        console.error("Failed to parse base64 audio:", error);
    }
}

async function sendMessage(overrideText = null) {
    if (!userInput) return;
    const textToSend = overrideText !== null ? overrideText : userInput.value.trim();
    if (!textToSend && audioChunks.length === 0) return;

    if (overrideText === null) {
        appendMessage(textToSend, "user");
        userInput.value = "";
    }

    let payload = { session_id: sessionId, message: textToSend, audio_base64: null };

    if (!textToSend && audioChunks.length > 0) {
        const loadingDiv = document.createElement("div");
        loadingDiv.className = "message bot loading-msg";
        loadingDiv.textContent = "🎙️ Transcribing voice note...";
        chatMessages.appendChild(loadingDiv);
        chatMessages.scrollTop = chatMessages.scrollHeight;

        const audioBlob = new Blob(audioChunks, { type: 'audio/webm' });
        const reader = new FileReader();
        reader.readAsDataURL(audioBlob);
        reader.onloadend = async () => {
            payload.audio_base64 = reader.result.split(',')[1];
            audioChunks = []; 
            await executeChatRequest(payload, loadingDiv);
        };
        return;
    }

    const loadingDiv = document.createElement("div");
    loadingDiv.className = "message bot loading-msg";
    loadingDiv.textContent = "Trackmate is thinking...";
    chatMessages.appendChild(loadingDiv);
    chatMessages.scrollTop = chatMessages.scrollHeight;

    await executeChatRequest(payload, loadingDiv);
}

async function executeChatRequest(payload, loadingDiv) {
    try {
        const response = await fetch("/chat", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(payload)
        });

        if (!response.ok) throw new Error(`Server Error: ${response.status}`);
        const data = await response.json();
        
        if (loadingDiv) loadingDiv.remove();
        if (data.audio) playAudio(data.audio);
        appendMessage(data.response, "bot");

        if (data.tool_triggered && data.tool_triggered.length > 0) {
            if (typeof window.searchPlace === "function") {
                data.tool_triggered.forEach(place => {
                    const searchBox = document.getElementById("mapSearchBox");
                    if (searchBox) searchBox.value = place;
                    window.searchPlace();
                });
            }
        }
    } catch (error) {
        if (loadingDiv) loadingDiv.remove();
        console.error("Fetch failure:", error);
        appendMessage("An issue occurred. Please check your connection.", "bot");
    }
}

const weatherApiKey = 'fc64e8cbf84d5060f7d160fa3a63acaf';

// Function to fetch and update weather metrics inside the upper right screen quadrant
function fetchLocalAppWeather(lat, lon) {
    const apiUrl = `https://api.openweathermap.org/data/2.5/weather?lat=${lat}&lon=${lon}&appid=${weatherApiKey}&units=metric`;
    
    fetch(apiUrl)
        .then(response => {
            if (!response.ok) throw new Error("Weather coordinates unreachable.");
            return response.json();
        })
        .then(data => {
            const temp = Math.round(data.main.temp);
            const conditionCode = data.weather[0].icon;
            
            // Map the API string weather icons over to sleek emoji elements
            let conditionEmoji = "☀️";
            if (conditionCode.includes("02") || conditionCode.includes("03") || conditionCode.includes("04")) {
                conditionEmoji = "⛅";
            } else if (conditionCode.includes("09") || conditionCode.includes("10")) {
                conditionEmoji = "🌧️";
            } else if (conditionCode.includes("11")) {
                conditionEmoji = "⚡";
            } else if (conditionCode.includes("13")) {
                conditionEmoji = "❄️";
            } else if (conditionCode.includes("50")) {
                conditionEmoji = "🌫️";
            }

            // Update UI elements smoothly
            document.getElementById("localWeatherIcon").innerText = conditionEmoji;
            document.getElementById("localWeatherTemp").innerText = `${temp}°C`;
            document.getElementById("localWeatherPill").title = `Current Location: ${data.name} (${data.weather[0].description})`;
        })
        .catch(err => {
            console.error("Weather Pipeline Error:", err.message);
            document.getElementById("localWeatherIcon").innerText = "⚠️";
        });
}

const trainLinePath = [
    [22.5830, 88.3425], [22.5802, 88.2774], [22.3892, 87.7121], [22.3330, 87.3250],

function getNearestStation(lat, lng) {
    let stations = chaibasaPlaces.filter(p => p.category === "Train Station");
    let nearest = null, minDistance = Infinity;
    stations.forEach(station => {
        let dist = L.latLng(lat, lng).distanceTo(L.latLng(station.lat, station.lng)) / 1000;
        if (dist < minDistance) { minDistance = dist; nearest = station; }
    });
    return minDistance <= 40 ? { station: nearest, distance: minDistance } : null; 
}

function getTrainPathSegment(startLat, startLng, endLat, endLng) {
    let startIndex = 0, minStartDist = Infinity, endIndex = 0, minEndDist = Infinity;
    
    trainLinePath.forEach((pt, idx) => {
        let dStart = L.latLng(startLat, startLng).distanceTo(L.latLng(pt[0], pt[1]));
        let dEnd = L.latLng(endLat, endLng).distanceTo(L.latLng(pt[0], pt[1]));
        if(dStart < minStartDist) { minStartDist = dStart; startIndex = idx; }
        if(dEnd < minEndDist) { minEndDist = dEnd; endIndex = idx; }
    });
    
    let path = (startIndex <= endIndex) ? trainLinePath.slice(startIndex, endIndex + 1) : trainLinePath.slice(endIndex, startIndex + 1).reverse();
    if (startLat === 22.6784 || endLat === 22.6784) path.push([22.6784, 85.6263]);
    path.unshift([startLat, startLng]); path.push([endLat, endLng]);
    return path;
}

var isTrainViewActive = false;
var activeTrainSegment = null;
var firstMileLine = null;

window.toggleDynamicTrainSchedule = function(sLat, sLng, eLat, eLng, startLat, startLng, endLat, endLng) {
    const ts = document.getElementById('dynamicTrainSchedule');
    const tbt = document.getElementById('turnByTurnUI');
    
    if (!isTrainViewActive) {
        isTrainViewActive = true;
        if(ts) ts.style.display = 'block';
        if(tbt) tbt.style.display = 'none';

        routingControl.setWaypoints([]); 
        
        let segmentCoords = getTrainPathSegment(sLat, sLng, eLat, eLng);

        lastMileLine = L.polyline([[eLat, eLng], [endLat, endLng]], { color: '#34495e', weight: 4, dashArray: '8, 8' }).addTo(map);
        map.fitBounds(activeTrainSegment.getBounds(), { padding: [50, 50] });

    } else {
        isTrainViewActive = false;
        if(ts) ts.style.display = 'none';
        if(tbt) tbt.style.display = 'block'; 

        if(activeTrainSegment) map.removeLayer(activeTrainSegment);
        if(firstMileLine) map.removeLayer(firstMileLine);
        if(lastMileLine) map.removeLayer(lastMileLine);
        routingControl.setWaypoints([L.latLng(startLat, startLng), L.latLng(endLat, endLng)]);
    }

async function getTrainSchedule() {
    const trainNo = document.getElementById("trainNo").value.trim();

    if (!trainNo) {
        alert("Enter Train Number");
        return;
    }

    // Checking our local database object
    const trainData = localTrainDatabase[trainNo];

    if (!trainData) {
        alert(`ℹ️ Train ${trainNo} is not added to the local data system yet.`);
        return;
    }

    // Clear previous dashboard items
    const tbody = document.querySelector("#scheduleTable tbody");
    tbody.innerHTML = "";

    // Set Header Card
    document.getElementById("trainInfo").innerHTML = `
        <strong>Train Number:</strong> ${trainData.train_number} <br>
        <strong>Train Name:</strong> ${trainData.train_name}
    `;

    const stations = trainData.route || [];

    if (stations.length === 0) {
        tbody.innerHTML = `<tr><td colspan="7">No route schedule available.</td></tr>`;
        return;
    }

    // Build table rows smoothly
    let tableRows = "";
    stations.forEach((station, index) => {
        tableRows += `
            <tr>
                <td>${index + 1}</td>
                <td>${station.station_name || "-"}</td>
                <td>${station.station_code || "-"}</td>
                <td>${station.sta || "-"}</td>
                <td>${station.std || "-"}</td>
                <td>${station.day || "1"}</td>
                <td>${station.distance || "0"}</td>
            </tr>
        `;
    });

    tbody.innerHTML = tableRows;
}

async function getTrainSchedule() {
    const trainNo = document.getElementById("trainNo").value.trim();

    if (!trainNo) {
        alert("Enter Train Number");
        return;
    }

    // Checking our local database object
    const trainData = localTrainDatabase[trainNo];

    if (!trainData) {
        alert(`ℹ️ Train ${trainNo} is not added to the local data system yet.`);
        return;
    }

    // Clear previous dashboard items
    const tbody = document.querySelector("#scheduleTable tbody");
    tbody.innerHTML = "";

    // Set Header Card
    document.getElementById("trainInfo").innerHTML = `
        <strong>Train Number:</strong> ${trainData.train_number} <br>
        <strong>Train Name:</strong> ${trainData.train_name}
    `;

    const stations = trainData.route || [];

    if (stations.length === 0) {
        tbody.innerHTML = `<tr><td colspan="7">No route schedule available.</td></tr>`;
        return;
    }

    // Build table rows smoothly
    let tableRows = "";
    stations.forEach((station, index) => {
        tableRows += `
            <tr>
                <td>${index + 1}</td>
                <td>${station.station_name || "-"}</td>
                <td>${station.station_code || "-"}</td>
                <td>${station.sta || "-"}</td>
                <td>${station.std || "-"}</td>
                <td>${station.day || "1"}</td>
                <td>${station.distance || "0"}</td>
            </tr>
        `;
    });

    tbody.innerHTML = tableRows;
}
